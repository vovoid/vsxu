cmake_minimum_required(VERSION 2.8)

include(${CMAKE_SOURCE_DIR}/cmake/CMakeFindLib.txt)

  add_definitions(
    -g
   -Wtype-limits
    -Wuninitialized
    -Wignored-qualifiers
    -Wempty-body
    -Wunused-parameter

endif()

  ${CMAKE_SOURCE_DIR}/lib/common/include
